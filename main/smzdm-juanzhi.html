    def calculate_price_per_gram(info):
  """
  根据商品信息计算卷纸的每克单价。

  Args:
    info: 包含卷纸信息的字符串。

  Returns:
    卷纸的每克单价，格式为 "X.XX 元/g"。
  """
  import re

  price_match = re.search(r"(\d+\.?\d*) 元", info)
  if not price_match:
    return "无法提取价格"

  price = float(price_match.group(1))

  gram_match = re.search(r"(\d+)g(\d+)\s*卷", info)
  if gram_match:
    grams = int(gram_match.group(1)) * int(gram_match.group(2))
  else:
    gram_match = re.search(r"(\d+)g", info)
    if gram_match:
      grams = int(gram_match.group(1))
    else:
      return "无法提取克数"

  price_per_gram = price / grams
  return f"{price_per_gram:.6f} 元/g"
